name: Build

on:
  push:
    branches:
      - main
      - develop
      - feature/v0.2.9-sonar-ci-restore
  pull_request: # Adicionado para validar código antes do merge
    branches:
      - main

jobs:
  build:
    name: Build and analyze
    runs-on: ubuntu-latest
    
    steps:
      - name: Checkout code
        uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: Set up JDK 21 # Ajustado para sua realidade (Java 21)
        uses: actions/setup-java@v4
        with:
          java-version: '21'
          distribution: 'temurin'
          cache: maven # Substitui a necessidade de blocos manuais de cache m2

      - name: Cache SonarQube packages
        uses: actions/cache@v4
        with:
          path: ~/.sonar/cache
          key: ${{ runner.os }}-sonar
          restore-keys: ${{ runner.os }}-sonar

      - name: Build and analyze
        env:
          SONAR_TOKEN: ${{ secrets.SONAR_TOKEN }}
          SONAR_HOST_URL: ${{ secrets.SONAR_HOST_URL }}
          SONAR_ORGANIZATION: ${{ secrets.SONAR_ORGANIZATION }}
        run: |
          mvn -B clean verify sonar:sonar \
          -Dsonar.projectKey=cloud-wallet-service-api \
          -Dsonar.projectName=wallet-service-api \
          -Dsonar.organization='${SONAR_ORGANIZATION}' \
          -Dsonar.coverage.jacoco.xmlReportPaths=target/site/jacoco/jacoco.xml
